```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* getKthFromEnd(ListNode* head, int k) {
        int length=0;
        ListNode* p=head;
        while(p->next!=NULL){
            length++;
            p=p->next;
        }

        int index=length-k+1;

        p=head;
        int j=0;
        while(j<index){
            p=p->next;
            j++;
        }

        return p;
    }
};
```
